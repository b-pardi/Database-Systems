CREATE TABLE General (
    city_state VARCHAR(20) NOT NULL PRIMARY KEY,
    county VARCHAR(20) NOT NULL,
    populations INTEGER(10) NOT NULL,
    pop_density INTEGER(6) NOT NULL,
    time_zone VARCHAR(20) NOT NULL,
    zip_codes VARCHAR(1000)
);

CREATE TABLE AQI (
    latitude DECIMAL(8,5) NOT NULL,
    longitude DECIMAL(8,5) NOT NULL,
    city_state VARCHAR(20) NOT NULL
    Time_updated datetime,
    AQI INTEGER(3) NOT NULL,
    CO INTEGER(4),
    H INTEGER(4),
    NO2 INTEGER(4),
    Ozone INTEGER(4),
    PM10 INTEGER(4),
    PM25 INTEGER(4)
    PRIMARY KEY (latitude, longitude)
);

CREATE TABLE AQI_Forecast(
    aqi_date datetime PRIMARY KEY
    city_state VARCHAR(20) NOT NULL,
    time_updated datetime,
    AQI INTEGER(3) NOT NULL,
    PM10 INTEGER(4),
    PM25 INTEGER(4)
);

CREATE TABLE Current_Weather(
    avg_coord DECIMAL(8,5) PRIMARY KEY,
    x_y_grid_pt VARCHAR(20) NOT NULL,
    city_state VARCHAR(20) NOT NULL,
    day_temp INTEGER(3) NOT NULL,--change to current_temp
    day_wind INTEGER(3) NOT NULL,--change to wind
    day_descr INTEGER(3) NOT NULL,--change to description
    night_temp INTEGER(3) NOT NULL,--remove
    night_wind INTEGER(3) NOT NULL,--remove
    night_descr INTEGER(3) NOT NULL--remove
);

CREATE TABLE Forecasted_Weather(
    weather_date datetime PRIMARY KEY --change to weather date location
    avg_coord DECIMAL(8,5) NOT NULL,
    city_state VARCHAR(20) NOT NULL,
    day_temp INTEGER(3) NOT NULL,--change to temperature
    day_wind INTEGER(3) NOT NULL,--change to wind
    day_descr INTEGER(3) NOT NULL,--change to description
    night_temp INTEGER(3) NOT NULL,--remove
    night_wind INTEGER(3) NOT NULL,--remove
    night_descr INTEGER(3) NOT NULL--remove
);

CREATE TABLE Water_Quality(
    usgs_id INTEGER(10) NOT NULL PRIMARY KEY,
    city_state VARCHAR(20) NOT NULL UNIQUE,
    hardness INTEGER(3),
    ph FLOAT DECIMAL(4,2),
    conductance INTEGER(5),
    ammonia FLOAT(4,2),
    nitrites FLOAT(5,3),
    fluoride DECIMAL(4,3),
    arsenic DECIMAL(4,2),
    cobalt DECIMAL(5,2),
    iron DECIMAL(6,2),
    lead DECIMAL(5,3),
    mercury DECIMAL(7,3)
);
